01: /**
02:  * 习题 7.10.3 的实现示例:
03:  *  对于来自命令行参数中的每个名字，均显示 "Hello xxx!"
04:  */
05: 
06: #include <iostream>
07: 
08: using namespace std;
09: 
10: /**
11:  * @brief 主程序
12:  * @param argc  命令行参数的个数，最小值为1
13:  * @param argv  命令行参数列表，是一个指针数组，元素个数为 argc
14:  *      其中 argv[0] 为可执行程序(*.exe)的路径.
15:  * @return 若在命令行参数中未指定任何名字则返回1，否则返回0.
16:  */
17: int main(int argc, char* argv[])
18: {
19:     if(argc == 1)
20:     {
21:         cerr << "Usage:\n    " << argv[0] << " name1 [name2 ...]\n";
22:         return 1;
23:     }
24: 
25:     for(int i=1; i<argc; ++i)
26:         cout << "Hello " << argv[i] << "!" << endl;
27:     return 0;
28: }
